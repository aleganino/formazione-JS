 function Dog(name) {
  this.name = name;
}

Dog.prototype.numLegs = 4;

let beagle = new Dog("Snoopy");

let ownProps = [];
let prototypeProps = [];

// Modifica il codice solo sotto questa riga
for (let property in beagle) {
  if (beagle.hasOwnProperty(property))
    ownProps.push(property);
  else
    prototypeProps.push(property);
}
